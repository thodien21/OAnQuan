	/// <summary>
        /// Etablish a new board/ Or recover the saved board
        /// </summary>
        /// <param name="isSavedOne">new Board = 0, saved board = 1</param>
        public Board(NoveltyOfGame novelty)
        {
            SquaresList = new List<Square>();
            ClickedSquares = new List<int>();
            PlayersList = new List<Player>();

            //Affect player number to each small square
            int n = 1;
            for (int k = 1; k <= PlayerQty; k++)
            {
                for (int i = n; i < n + 5; i++)
                {
                    SquaresList[i].PlayerNumber = k;
                }
                n = n + 6;
            }

            //Set up the player list of board
            if(novelty == NoveltyOfGame.NEW)
            {
                PlayersList.Add(new Player(Services.Player.Pseudo));
                PlayersList.Add(new Player(Player2Pseudo));

                Turn = new Random().Next(1, 3); //Board decides who plays first (1 or 2).

                //
                for (int i = 0; i < PlayerQty; i++)
                {
                    SquaresList[i].BigTokenQty = 1;
                    for (int j = 0; j < 5; j++)
                    {
                        SquaresList.Add(new SmallSquare());
                    }
                }
            }
            

            for(int i=0; i<PlayerQty; i++)
            {
                SquaresList.Add(new BigSquare());
                for (int j = 0; j < 5; j++)
                {
                    SquaresList.Add(new SmallSquare());
                }
            }
        }
